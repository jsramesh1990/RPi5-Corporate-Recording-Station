Running Tests
bash

# Build and run all integration tests
cd build
make test-integration

# Run specific test suite
./tests/integration/test_full_system
./tests/integration/test_recording_pipeline
./tests/integration/test_uart

# Run with verbose output
./tests/integration/test_full_system --gtest_verbose=1

# Run specific test case
./tests/integration/test_recording_pipeline --gtest_filter=RecordingPipelineTest.VideoEncoding
