Usage Example
cpp

// Initialize camera
CameraInterface camera;
CameraInterface::CameraConfig config;
config.device = "/dev/video0";
config.width = 1920;
config.height = 1080;
config.framerate = 30;
camera.initialize(config);

// Set frame callback
camera.set_on_frame([](const CameraInterface::FrameInfo& frame) {
    // Process frame
    std::cout << "Frame: " << frame.width << "x" << frame.height 
              << ", " << frame.data.size() << " bytes" << std::endl;
});

// Start streaming
camera.start_streaming();

// ... streaming ...

// Capture a single frame
auto frame = camera.capture_frame();
camera.save_frame(frame, "test.jpg");

// Stop streaming
camera.stop_streaming();

// Cleanup
camera.shutdown();
