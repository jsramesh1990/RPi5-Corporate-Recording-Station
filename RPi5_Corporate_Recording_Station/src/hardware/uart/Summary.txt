Usage Examples
cpp

// Initialize UART
PL011UART uart;
PL011UART::UARTConfig config;
config.device = "/dev/ttyAMA0";
config.baud_rate = 115200;
uart.initialize(config);

// Write data
uart.writeData("Hello UART!\r\n");

// Read data
auto data = uart.readData(100);

// Async read
uart.startAsyncRead([](const std::vector<uint8_t>& data) {
    std::cout << "Received: " << std::string(data.begin(), data.end());
});

// UART Console
UARTConsole console;
console.initialize("/dev/ttyAMA0", 115200);
console.setPrompt("record> ");
console.setCommandHandler([](const std::string& cmd) {
    if (cmd == "status") return "System: Online";
    if (cmd == "record") return "Recording started";
    if (cmd == "stop") return "Recording stopped";
    return "Unknown command";
});
console.enable();
