Dashboard Renderer Summary
File	Purpose	Key Features
DashboardRenderer.hpp	Dashboard header	Types, structures, callbacks, public API
DashboardRenderer.cpp	Dashboard implementation	Multi-mode rendering, data visualization, export
Key Features

    Data Types:

        SystemMetrics: CPU, memory, temperature, load, uptime

        RecordingStatus: Recording state, file, duration, size, codecs

        StorageStatus: Total, used, free, file count, type distribution

        CloudStatus: Connection, sync, provider, space usage

        NetworkStatus: Interface, IP, traffic, WiFi

        GPIOStatus: Pin states, direction, pull, change count

    Rendering Modes:

        FULL: All sections

        MINIMAL: Essential info only

        RECORDING: Focus on recording

        MONITORING: Focus on system health

        STORAGE: Focus on storage management

        NETWORK: Focus on network status

    Visual Features:

        Color-coded output

        Progress bars

        Icons for status

        Compact mode

        Real-time updates

    Export:

        Plain text export

        HTML export

        File export

Usage Example
cpp

// Create dashboard
DashboardRenderer dashboard;
dashboard.initialize("Recording Station");

// System metrics
DashboardRenderer::SystemMetrics metrics;
metrics.cpu_usage = 25.5;
metrics.cpu_temp = 45.2;
metrics.memory_total = 4096 * 1024 * 1024;
metrics.memory_used = 1536 * 1024 * 1024;
metrics.memory_free = 2560 * 1024 * 1024;
metrics.uptime_seconds = 86400;
metrics.hostname = "recording-station";
dashboard.updateSystemMetrics(metrics);

// Recording status
DashboardRenderer::RecordingStatus recording;
recording.is_recording = true;
recording.filename = "meeting_20260708.mp4";
recording.duration_seconds = 3600;
recording.file_size_bytes = 1500000000;
recording.fps = 30.0;
recording.video_codec = "H.264";
recording.audio_codec = "AAC";
dashboard.updateRecordingStatus(recording);

// Storage status
DashboardRenderer::StorageStatus storage;
storage.total_bytes = 128 * 1024 * 1024 * 1024;
storage.used_bytes = 76.8 * 1024 * 1024 * 1024;
storage.free_bytes = 51.2 * 1024 * 1024 * 1024;
storage.file_count = 1247;
storage.file_type_distribution = {{".mp4", 342}, {".wav", 156}, {".json", 89}};
dashboard.updateStorageStatus(storage);

// Cloud status
DashboardRenderer::CloudStatus cloud;
cloud.is_connected = true;
cloud.provider = "Nextcloud";
cloud.username = "recording_station";
cloud.total_space = 500 * 1024 * 1024 * 1024;
cloud.used_space = 200 * 1024 * 1024 * 1024;
cloud.free_space = 300 * 1024 * 1024 * 1024;
dashboard.updateCloudStatus(cloud);

// Network status
DashboardRenderer::NetworkStatus network;
network.interface = "eth0";
network.ip_address = "192.168.1.100";
network.is_connected = true;
network.bytes_sent = 1024 * 1024 * 1024;
network.bytes_received = 2048 * 1024 * 1024;
dashboard.updateNetworkStatus(network);

// Logs
dashboard.addLog("System started", "INFO");
dashboard.addLog("Recording started: meeting_20260708.mp4", "SUCCESS");
dashboard.addLog("Storage usage at 60%", "WARNING");
dashboard.addLog("Failed to upload file", "ERROR");

// Set mode
dashboard.setMode(DashboardRenderer::DashboardMode::FULL);

// Render
dashboard.render();

// Export
std::string html = dashboard.exportToHTML();
dashboard.exportToFile("dashboard.html");
