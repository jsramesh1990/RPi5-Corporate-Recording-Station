Deployment Commands
bash

# Start all services
cd config/docker
docker-compose up -d

# Check status
docker-compose ps

# View logs
docker-compose logs -f recording-station

# Stop all services
docker-compose down

# Update and restart
docker-compose pull
docker-compose up -d --build

# Clean up
docker-compose down -v
